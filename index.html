-- "which applicants missed their third installment?"
SELECT a.id, a.name, a.email
FROM   applicants a
JOIN   installments i ON i.applicant_id = a.id
WHERE  i.sequence = 3
  AND  i.paid_at IS NULL
  AND  i.due_at < now()
ORDER BY i.due_at DESC
LIMIT 50;
